#include <stdio.h>
#include <stdlib.h>

int ifOdd(int **arr,int n){
  int sum = 0;
    for (int i = 0,j=0; i < n && j<n; i++,j++) {
    sum = sum + arr[i][i];
    sum = sum + arr[n - 1 - j][n - 1 - i];
  }
  sum = sum - arr[n / 2][n / 2];
  return sum;
}

int ifEven(int **arr,int n){
  int sum = 0;
  for (int i = 0,j=0; i < n && j<n; i++,j++) {
      sum = sum + arr[i][i];
    sum = sum + arr[n - 1 - j][n - 1 - i];
    }
  return sum;
}

int main() {
  printf("enter no of rows and columns\n");
  int n;
  scanf("%d", &n);
  int **arr = (int **)malloc(n * sizeof(int *));
  for (int i = 0; i < n; i++){
    arr[i] = (int *)malloc(n * sizeof(int));
  }
  int result;
  printf("Enter the Matrix\n");
  for (int i = 0; i < n; i++) {
    for (int j = 0; j < n; j++) {
      scanf("%d", &arr[i][j]);
    }
  }
  if (n % 2 == 0) {
    result = ifEven(arr, n);
  } else {
    result = ifOdd(arr, n);
  }
  printf("%d", result);
  return 0;
}
